tree("Root")
	fallback
		tree("Run away")
		tree("Need food")
		tree("Need water")
		tree("Need to breed")
		tree("Move randomly")

tree("Run away")
	while IsInDanger
		RunAway

tree("Need food")
	while IsHungry
		mute
			sequence
				SearchForFood
				GoToFood
				Wait(2.0)
				EatFood

tree("Need water")
	while IsThirsty
		mute
			sequence
				GoToWater
				Wait(2.0)
				DrinkWater

tree("Need to breed")
	while
		sequence
			IsOldEnoughForSex
			not IsPregnant

		mute
			sequence
				SearchForAPartner
				GoToPartner
				Wait(2.0)
				sequence
					IsMale
					ImpregnateMate

tree("Move randomly")
	MoveRandomly
